JavaScript基础语法详解

2024-01-05 JavaScript 基础语法 128

什么是JavaScript？

JavaScript是一种轻量级的脚本语言，主要用于网页交互和动态内容生成。它是一种解释型语言，不需要编译，可以直接在浏览器中运行。

JavaScript最初是为了给网页添加交互效果而设计的，但现在已经发展成为一种功能强大的编程语言，可以用于开发桌面应用、移动应用、服务器端应用等。

JavaScript的基本语法

1. 变量声明

在JavaScript中，我们可以使用var、let或const关键字来声明变量：

// 使用var声明变量（ES5及之前）
var name = "HTMLer";

// 使用let声明变量（ES6+）
let age = 25;

// 使用const声明常量（ES6+）
const PI = 3.14159;

三者的主要区别：

  • var：函数作用域，存在变量提升
  • let：块级作用域，不存在变量提升
  • const：块级作用域，声明后不可修改

2. 数据类型

JavaScript有七种基本数据类型和一种引用数据类型：

基本数据类型

  • String：字符串，如"Hello World"
  • Number：数字，如123、3.14
  • Boolean：布尔值，true或false
  • Null：空值，表示一个空对象指针
  • Undefined：未定义，表示变量未被赋值
  • Symbol：唯一标识符（ES6+）
  • BigInt：大整数（ES2020+）

引用数据类型

  • Object：对象，包括数组、函数、日期等
// 基本数据类型
let str = "JavaScript";  // String
let num = 100;           // Number
let bool = true;         // Boolean
let n = null;            // Null
let undef;               // Undefined
let sym = Symbol("id");  // Symbol
let big = 123n;          // BigInt

// 引用数据类型
let obj = {name: "HTMLer", age: 25};  // 对象
let arr = [1, 2, 3, 4, 5];             // 数组
let func = function() {console.log("Hello");};  // 函数

3. 运算符

算术运算符

let a = 10;
let b = 3;

console.log(a + b);  // 13
console.log(a - b);  // 7
console.log(a * b);  // 30
console.log(a / b);  // 3.3333333333333335
console.log(a % b);  // 1
console.log(a++);    // 10，先返回a的值，再自增
console.log(++a);    // 12，先自增，再返回a的值

比较运算符

let a = 10;
let b = "10";

console.log(a == b);   // true，相等比较，会进行类型转换
console.log(a === b);  // false，严格相等比较，不会进行类型转换
console.log(a != b);   // false
console.log(a !== b);  // true
console.log(a > b);    // false
console.log(a < b);    // false
console.log(a >= b);   // true
console.log(a <= b);   // true

逻辑运算符

let a = true;
let b = false;

console.log(a && b);  // false，逻辑与
console.log(a || b);  // true，逻辑或
console.log(!a);      // false，逻辑非

4. 流程控制语句

条件语句

// if-else语句
let score = 85;

if (score >= 90) {
    console.log("优秀");
} else if (score >= 80) {
    console.log("良好");
} else if (score >= 60) {
    console.log("及格");
} else {
    console.log("不及格");
}

// switch语句
let day = 3;
let dayName;

switch (day) {
    case 1:
        dayName = "星期一";
        break;
    case 2:
        dayName = "星期二";
        break;
    case 3:
        dayName = "星期三";
        break;
    default:
        dayName = "未知";
}

循环语句

// for循环
for (let i = 0; i < 5; i++) {
    console.log(i);  // 输出0, 1, 2, 3, 4
}

// while循环
let i = 0;
while (i < 5) {
    console.log(i);  // 输出0, 1, 2, 3, 4
    i++;
}

// do-while循环
let j = 0;
do {
    console.log(j);  // 输出0, 1, 2, 3, 4
    j++;
} while (j < 5);

// for-in循环（遍历对象属性）
let person = {name: "HTMLer", age: 25};
for (let key in person) {
    console.log(key + ": " + person[key]);  // name: HTMLer, age: 25
}

// for-of循环（遍历可迭代对象）
let arr = [1, 2, 3, 4, 5];
for (let value of arr) {
    console.log(value);  // 输出1, 2, 3, 4, 5
}

5. 函数

函数是JavaScript中的基本构建块，用于封装可重用的代码块：

// 函数声明
function add(a, b) {
    return a + b;
}

// 函数表达式
let subtract = function(a, b) {
    return a - b;
};

// 箭头函数（ES6+）
let multiply = (a, b) => a * b;

// 调用函数
console.log(add(10, 5));       // 15
console.log(subtract(10, 5));  // 5
console.log(multiply(10, 5));  // 50

6. 数组

数组是一种特殊的对象，用于存储多个值：

// 创建数组
let fruits = ["苹果", "香蕉", "橙子"];
let numbers = new Array(1, 2, 3, 4, 5);

// 访问数组元素
console.log(fruits[0]);  // 苹果

// 修改数组元素
fruits[1] = "葡萄";

// 数组方法
fruits.push("草莓");          // 添加元素到数组末尾
fruits.pop();                // 移除数组末尾的元素
fruits.unshift("芒果");       // 添加元素到数组开头
fruits.shift();              // 移除数组开头的元素
fruits.splice(1, 0, "梨");    // 在指定位置插入元素
fruits.splice(1, 1);         // 移除指定位置的元素

// 数组遍历
fruits.forEach(function(fruit) {
    console.log(fruit);
});

// 数组转换
let str = fruits.join(", ");  // 苹果, 梨, 橙子
let newArr = fruits.map(function(fruit) {
    return fruit + "好吃";
});

7. 对象

对象是JavaScript中的核心概念，用于存储键值对：

// 创建对象
let person = {
    name: "HTMLer",
    age: 25,
    sayHello: function() {
        console.log("Hello, I'm " + this.name);
    }
};

// 访问对象属性
console.log(person.name);       // HTMLer
console.log(person["age"]);     // 25

// 修改对象属性
person.age = 26;

// 添加新属性
person.gender = "男";

// 删除属性
delete person.gender;

// 调用对象方法
person.sayHello();  // Hello, I'm HTMLer

JavaScript的基本概念

1. 作用域

作用域是指变量和函数的可访问范围：

  • 全局作用域：在所有函数外部声明的变量，可在整个程序中访问
  • 函数作用域：在函数内部声明的变量，只能在函数内部访问
  • 块级作用域：在{}内部声明的变量，只能在块内部访问（ES6+）

2. 闭包

闭包是指有权访问另一个函数作用域中变量的函数：

function outer() {
    let count = 0;
    
    function inner() {
        count++;
        console.log(count);
    }
    
    return inner;
}

let counter = outer();
counter();  // 1
counter();  // 2
counter();  // 3

3. 原型和继承

JavaScript使用原型链实现继承：

// 构造函数
function Person(name, age) {
    this.name = name;
    this.age = age;
}

// 添加方法到原型
Person.prototype.sayHello = function() {
    console.log("Hello, I'm " + this.name);
};

// 创建实例
let person1 = new Person("HTMLer", 25);
person1.sayHello();  // Hello, I'm HTMLer

// 继承
function Student(name, age, grade) {
    Person.call(this, name, age);
    this.grade = grade;
}

Student.prototype = Object.create(Person.prototype);
Student.prototype.constructor = Student;

let student = new Student("小明", 18, "高三");
student.sayHello();  // Hello, I'm 小明

4. 异步编程

JavaScript是单线程语言，但可以通过异步编程处理耗时操作：

// 回调函数
function fetchData(callback) {
    setTimeout(function() {
        callback("数据获取成功");
    }, 1000);
}

fetchData(function(data) {
    console.log(data);
});

// Promise（ES6+）
function fetchData() {
    return new Promise(function(resolve, reject) {
        setTimeout(function() {
            resolve("数据获取成功");
        }, 1000);
    });
}

fetchData().then(function(data) {
    console.log(data);
});

// async/await（ES2017+）
async function getData() {
    let data = await fetchData();
    console.log(data);
}

getData();

JavaScript的最佳实践

  • 使用let和const代替var
  • 遵循一致的命名约定
  • 使用箭头函数简化回调函数
  • 避免全局变量污染
  • 使用模块化组织代码
  • 添加适当的注释
  • 使用严格模式（'use strict';）
  • 进行适当的错误处理
  • 优化性能，避免不必要的DOM操作
  • 使用现代JavaScript特性

总结

JavaScript是一种功能强大的脚本语言，具有灵活的语法和广泛的应用场景。掌握JavaScript的基础语法是学习前端开发的重要一步。

本文详细介绍了JavaScript的基本语法，包括变量声明、数据类型、运算符、流程控制语句、函数、数组、对象、事件处理和DOM操作等内容。通过学习这些基础知识，你可以开始编写简单的JavaScript程序，并逐步深入学习更高级的概念。

JavaScript的学习是一个持续的过程，建议多实践、多阅读优秀的代码，并关注JavaScript的最新发展。
